1 function SetData(process) {
  2   
  3     fCodeCategory     = document.frmSearch.CodeCategory;
  4     fCodeCategoryAN = document.frmSearch.CodeCategoryAN;    // Categories related to Analysis
  5     fCodeCategoryAC = document.frmSearch.CodeCategoryAC;    // Categories related to Acquisition
  6     
  7     if (process == "Analysis") {
  8         document.frmSearch.process.value="AN";
  9         document.frmSearch.totalTools.value = document.frmSearch.nToolsAN.value;
 10         selectLength = fCodeCategoryAN.length;

 14     else {
 15         document.frmSearch.process.value="AC";

 18         fSelect = fCodeCategoryAC;
 19         document.frmSearch.dfTools.value = "Acquisition";
 20     }        
 21     
 22     fCodeCategory.length = selectLength
 23     for (i=0; i < selectLength; i++) {
 24         fCodeCategory.options[i].value = fSelect.options[i].value;

 26     }

 30     document.frmSearch.os.selectedIndex = 0;

 32     
 33

 37 function CheckCategory() {
 38     
 39     fProcess                    = document.frmSearch.process;
 40     fCategoryName         = document.frmSearch.Category;
 41     fCategories               = document.frmSearch.CodeCategory; 
 42     maxValuesFeature     = 4     // if a Feature has more then maxValuesFeature, then in the Features Panel are show two values for each row
 43     
 44     
 45     if (fProcess.value == "AN")         // case Analysis
 46         fFeatures                  = document.frmSearch.FeaturesAN;
 47     else                 // case Acquisition
 48         fFeatures                  = document.frmSearch.FeaturesAC;
 49     
 50     fFeaturesValues        = document.frmSearch.FeaturesValues;
 51     //fSubFeaturesValues  = document.frmSearch.SubFeatures;   
 52 
 53 
 54     idxCategory = fCategories.selectedIndex;
 55     vCategory = fCategories.options[idxCategory].value;
 56     idFeatures = fFeatures.options[idxCategory].value;
 57     nameFeatures = fFeatures.options[idxCategory].text;
 58     fCategoryName.value = fCategories.options[idxCategory].text;
 59     
 60     //window.alert("idx=" + idxCategory + "category=" + vCategory + ", \nidF=" + idFeatures + "\nnameF=" + nameFeatures);
 61 
 62     removePanel();
 63     
 64     if (idFeatures == "")  // if there isn't any feature  related to the chosen category
 65         ;
 66     else {   // there is, at least, a feature related to the chosen category, it shows the feature panel!
 67         divBox = $('<div id="box"></div>');
 68         divBox.appendTo("#container");
 69     }        
 70     
 71     aIdFeatures = idFeatures.split("#");                    // extract all features id separated by #
 72     aNamesFeatures = nameFeatures.split("#");      // extract all features names separated by #
 73     checkboxNumber = 0;
 74     for (k=0; k < aIdFeatures.length; k++) {            // loop on all Features related to the selected category
 75         idFeatureSubFeature = aIdFeatures[k].split("@");    // the value contains idFeature@DeeperLevel, DeeperLevel may assume S (with sub features) or N (without sub features)
 76         idFeature = idFeatureSubFeature[0];
 77         idSubFeature = idFeatureSubFeature[1];
 78         visibleFeature = idFeatureSubFeature[2]; 
 79         nameFeature = aNamesFeatures[k];
 80         valueFound = false;
 81         if (idSubFeature == "N") {                // feature without subfeatures
 82             for (i=0; i < fFeaturesValues.length; i++) {
 83                 if (fFeaturesValues.options[i].value == idFeature) {
 84                     checkboxNumber ++;
 85 //if the Feature is not visibile, case Category=03. and Feature=Function all values of that Feature are set on checked but the Feature is not shown on Features panel
 86                     if (visibleFeature =="N") {
 87                         classFeature  = " class=dftHidden ";
 88                         classRadio     = " class=dftHidden ";
 89                         sName = "";
 90                         checkboxNumber = 0;
 91                     }                        
 92                     else {
 93                         classFeature =" ";                    
 94                         classRadio = " class=dftTextGrassetto ";
 95                         sName = nameFeature
 96                     }                        
 97                            
 98                     if (checkboxNumber > 1)
 99                         labelRadio = "<hr><input type=checkbox " + classFeature + "checked name=" + idFeature;
100                     else
101                         labelRadio = "<input type=checkbox " + classFeature + "checked name=" + idFeature;    
102                         
103                     labelRadio = labelRadio + " onClick=javascript:SelectDeselectValues('" + idFeature + "');>";
104                     labelFeature = $(labelRadio + '<span ' + classRadio + '> ' + sName + '</span><br/>');
105                     labelFeature.appendTo("#box");
106                     aValues = fFeaturesValues.options[i].text.split("#");    // extract all values separated by #
107                     if (aValues.length > maxValuesFeature) { // if a Feature has more then maxValuesFeature, then in the Features Panel  two values for each row are shown
108                         for (j=0; j < aValues.length; j++) {
109                             nameCheckOne = idFeature + "_" + j;
110                             valueCheckOne = aValues[j];
111                             if (j == aValues.length - 1)
112                                 create_box_double_end(nameCheckOne, valueCheckOne, classRadio, sName); 
113                             else {
114                                 idx = j + 1;
115                                 nameCheckTwo = idFeature + "_" + idx;
116                                 valueCheckTwo = aValues[idx];
117                                 create_box_double (nameCheckOne, valueCheckOne, nameCheckTwo, valueCheckTwo, classRadio, sName);
118                                 j = j + 1;
119                             }                                                                
120                         }
121                     }
122                     else {
123                         for (j=0; j < aValues.length; j++) {
124                             nameCheck = idFeature + "_" + j;
125                             valueCheck = aValues[j];
126                             create_box_special(nameCheck, valueCheck, classRadio, sName); 
127                         }
128                     }                        
129                    valueFound = true;
130                    break;
131                 }
132             } 
133             if (!valueFound){
134                 labelFeature = $('<span class=dftTextGrassetto><input type=checkbox name=' + idFeature + ' value="Yes"> ' + nameFeature + '</span><br/>');
135                 labelFeature.appendTo("#box");
136             }
137         }
138     }        
139     
140    idx = document.frmSearch.CodeCategory.selectedIndex;
141 
142    if (document.frmSearch.process.value == "AN")
143         sizeBox = document.frmSearch.boxSizesAN.options[idx].value;
144     else
145         sizeBox = document.frmSearch.boxSizesAC.options[idx].value;               
146         
147     sizeContainer = (parseInt(sizeBox) + 30).toString();
148     $("#container").css("height",sizeContainer);
149     $("#box").css("height",sizeBox);
150     $("#box").toggle("slow");
151  }

153  function SelectDeselect(idFeature) {        // check box for selecting/deselecting all subfeature values
154     var elementsLength = document.frmSearch.elements.length;
155     //fSubFeatures = document.frmSearch.SubFeatures;
156     
157     lFound = false;
158     for (i=0; i < elementsLength; i++) {
159         if (frmSearch.elements[i].name == idFeature) {
160             fFeature = frmSearch.elements[i];
161             lFound = true;
162             break;
163         }            
164     }
165     if (lFound) {
166        for (j=0; j<fSubFeatures.length; j++)  {
167             if (fSubFeatures.options[j].value == fFeature.name) {
168                 //window.alert("before split #, " + fSubFeatures.options[j].text);
169                 valuesSubFeatures = fSubFeatures.options[j].text.split("#");
170                 for (k=0; k < valuesSubFeatures.length; k++) {
171                     idSubF = valuesSubFeatures[k].split("@");
172                     id = idSubF[1];
173                     for (m=0; m < elementsLength; m++) {
174                         if (frmSearch.elements[m].name == id) {
175                             fSubFeature = frmSearch.elements[m];     
176                             break;
177                         }
178                     }       
179                     if (fFeature.checked)
180                         fSubFeature.checked = true;
181                     else
182                         fSubFeature.checked = false;                                                
183                 }                    
184             }                
185         }           
186     }        
187 }

189  function SelectDeselectValues(idFeature) {            // check box for selecting/deselecting all feature values
190     var elementsLength = document.frmSearch.elements.length;
191     var maxValuesFeature = 20;        // there isn't more than 20 values for each feature
192     
193     lFound = false;
194     for (i=0; i < elementsLength; i++) {
195         if (frmSearch.elements[i].name == idFeature) {
196             fFeature = frmSearch.elements[i];
197             lFound = true;
198             break;
199         }            
200     }
201     if (lFound) {
202         value = fFeature.checked; 
203         //window.alert(value); 
204         for (j=0; j<maxValuesFeature; j++)  {
205             valueFeature = idFeature + "_" + j;
206             for (k=0; k < elementsLength; k++) {
207                 if (frmSearch.elements[k].name == valueFeature) {
208                     fValue = frmSearch.elements[k];     
209                     fValue.checked = value;
210                     break;      
211                 } 
212             }                                           
213         }           
214     }
